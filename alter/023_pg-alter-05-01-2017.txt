SET SCHEMA 'zafira';

DROP TRIGGER update_timestamp_test_runs ON zafira.test_runs;

--TEST_RUNS
DROP INDEX IF EXISTS TEST_RUNS_STARTED_AT_INDEX;
CREATE INDEX TEST_RUNS_STARTED_AT_INDEX ON TEST_RUNS (STARTED_AT);

UPDATE TEST_RUNS SET STARTED_AT = CREATED_AT WHERE STARTED_AT IS NULL;



CREATE TRIGGER update_timestamp_test_runs
  BEFORE INSERT OR UPDATE
  ON zafira.test_runs
  FOR EACH ROW
  EXECUTE PROCEDURE zafira.update_timestamp();
