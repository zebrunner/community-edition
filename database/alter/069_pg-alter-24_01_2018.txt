DO $$
DECLARE userId INT;
BEGIN
  FOR userId IN SELECT U.ID FROM zafira.USERS U LEFT JOIN zafira.USER_PREFERENCES UP ON U.ID = UP.USER_ID WHERE UP.ID IS NULL
  LOOP
    INSERT INTO zafira.USER_PREFERENCES(NAME, VALUE, USER_ID) VALUES
      ('REFRESH_INTERVAL', 0, userId),
      ('DEFAULT_DASHBOARD', 'General', userId),
      ('THEME', 32, userId);
  END LOOP;
END;
$$;
